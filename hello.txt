[hello].txt
# 폴더 만들기
mkdir hello

# 깃 로그 한글 깨짐
git config --global core.quotepath false

# 폴더 이동
cd hello

# 폴더의 목록 확인
ls -al

# 깃 저장소 생성: 최초에 한번만
git init

# 깃 상태 확인
git status

# 깃에 관래 대상으로 포함
# 변경사항이 있을 때마다 수행
git add hello.txt

# 로컬 저장소에 저장소
# 최종적으로 변경 사항을 저장소에 반영
git commit -m "[hello.txt] new file"

# 이력 상새 확인
git log

# 이력 간략 확인
git log --oneline

# hello.txt 파일의 복사본(hello.txt) 생성
cp hello,txt hello.txt
ls -al

# 현재 폴더(working directory)의 모든 내용을 반영
# 관리대상으로 만든다.
# 수정, 새로운 파일을 스테이지 영역으로 이동
git add .

# add와 commit을 동시 수행
# 조건: 한 번 관리 대상이 되었던 파일
# 옵션: -a
git commit -a -m "[hello.txt] 한글 깨지는 문제 수록"

# 브랜치 관리
# branch 생성
git branch 브랜치명
git branch br-1

# branch dlehd
git checkout br-1
git switch br-1

# branch 삭제
git branch -d br-1

# 브랜치 생성 및 이동
git checkout -b br-2
